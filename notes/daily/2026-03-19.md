```bash
python3 -m venv .venv

```

### 3. 创建 `app/main.py`

```bash
cat <<'EOF' > app/main.py
import os

from fastapi import FastAPI

from app.routers.health import router as health_router
from app.routers.users import router as users_router

app = FastAPI(title=os.getenv("APP_NAME", "all in ai"))

app.include_router(health_router)
app.include_router(users_router)
EOF
```

### 4. 创建 `app/routers/health.py`

```bash
cat <<'EOF' > app/routers/health.py
from fastapi import APIRouter

router = APIRouter()


@router.get("/health")
def health_check():
    return {"status": "ok"}
EOF
```

### 5. 创建 `app/routers/users.py`

```bash
cat <<'EOF' > app/routers/users.py
from fastapi import APIRouter
from pydantic import BaseModel

router = APIRouter(prefix="/users", tags=["users"])
users = []


class UserCreate(BaseModel):
    name: str


@router.get("/")
def list_users():
    return {"items": users}


@router.post("/")
def create_user(payload: UserCreate):
    user = {"id": len(users) + 1, "name": payload.name}
    users.append(user)
    return user
EOF
```

### 6. 用环境变量启动服务

```bash
APP_NAME="all in ai week1" uvicorn app.main:app --reload
```
